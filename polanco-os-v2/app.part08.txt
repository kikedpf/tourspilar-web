  function exportJSON() {
    const payload={version:2,exportedAt:new Date().toISOString(),settings:state.settings,operations:state.operations,compareIds:[...state.compareIds]};
    download(new Blob([JSON.stringify(payload,null,2)],{type:'application/json'}),`polanco_os_${today()}.json`);
  }

  function exportCSV() {
    const head=['Nombre','Barrio','Estado','Precio publicado','Compra','Venta','Área','Score','Precio máximo','Inversión','Utilidad','Margen neto','ROI','Comparables','Decisión','Actualizado'];
    const rows=state.operations.map(op=>{const r=calc(op);return [op.name,op.neighborhood,op.status,op.listedPrice,op.purchasePrice,op.salePrice,op.area,r.score.total,r.actual.maxPurchase,r.breakdown.cashCost,r.actual.netProfit,r.actual.netMargin,r.actual.roi,r.comparables.count,r.actual.decision,op.updatedAt]});
    const csv=[head,...rows].map(row=>row.map(v=>`"${String(v??'').replace(/"/g,'""')}"`).join(';')).join('\n');
    download(new Blob(['\ufeff'+csv],{type:'text/csv;charset=utf-8'}),`operaciones_polanco_${today()}.csv`);
  }

  function download(blob,name){const url=URL.createObjectURL(blob),a=document.createElement('a');a.href=url;a.download=name;a.click();setTimeout(()=>URL.revokeObjectURL(url),1200)}

  async function handleImport(event) {
    const file=event.target.files[0]; if(!file)return;
    try{
      const data=JSON.parse(await file.text());
      if(!Array.isArray(data.operations)) throw new Error('Copia no válida');
      state.operations=data.operations;
      state.settings={...defaultSettings(),...(data.settings||{})};
      state.compareIds=new Set(data.compareIds||[]);
      state.currentId=state.operations[0]?.id||null;
      state.draft=state.currentId?clone(state.operations[0]):baseOperation();
      persist();refreshOperationSelect();render();toast('Copia importada');
    }catch(error){toast(`No se pudo importar: ${error.message}`)}
    event.target.value='';
  }

  function copySummary() {
    const r=calc();
    const text=[`POLANCO OS · ${state.draft.name}`,`Compra: ${money(state.draft.purchasePrice)}`,`Venta: ${money(state.draft.salePrice)}`,`Precio máximo: ${money(r.actual.maxPurchase)}`,`Inversión: ${money(r.breakdown.cashCost)}`,`Utilidad neta: ${money(r.actual.netProfit)}`,`Margen neto: ${pct(r.actual.netMargin)}`,`ROI: ${pct(r.actual.roi)}`,`Score: ${r.score.total}/100`, `Decisión: ${r.actual.decision}`].join('\n');
    navigator.clipboard?.writeText(text).then(()=>toast('Resumen copiado')).catch(()=>prompt('Copia el resumen:',text));
  }
