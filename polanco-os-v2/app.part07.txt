  function bindDynamicAfterRender() {
    const photoInput=$('photo-input'); if(photoInput) photoInput.addEventListener('change',handlePhotos);
    const importInput=$('import-json'); if(importInput) importInput.addEventListener('change',handleImport);
    qsa('[data-drag-id]').forEach(card=>card.addEventListener('dragstart',e=>{e.dataTransfer.setData('text/plain',card.dataset.dragId);card.classList.add('dragging')}));
    qsa('[data-drag-id]').forEach(card=>card.addEventListener('dragend',()=>card.classList.remove('dragging')));
    qsa('[data-drop-status]').forEach(col=>{
      col.addEventListener('dragover',e=>{e.preventDefault();col.classList.add('drop-active')});
      col.addEventListener('dragleave',()=>col.classList.remove('drop-active'));
      col.addEventListener('drop',e=>{e.preventDefault();col.classList.remove('drop-active');movePipeline(e.dataTransfer.getData('text/plain'),col.dataset.dropStatus)});
    });
  }

  function movePipeline(id,status) {
    const op=state.operations.find(o=>o.id===id); if(!op)return;
    op.status=status; op.updatedAt=new Date().toISOString();
    if(state.currentId===id){state.draft.status=status;state.draft.updatedAt=op.updatedAt}
    persist();render();toast(`Movida a ${status}`);
  }

  function updateDraftFromElement(el) {
    const key=el.dataset.bind; if(!key||!state.draft)return;
    let value=el.value;
    if(el.dataset.kind==='number') value=Number(value)||0;
    if(el.dataset.kind==='pct') value=(Number(value)||0)/100;
    if(el.dataset.kind==='checkbox') value=el.checked;
    if(key==='financeCustomBase'){state.financeUI.customBase=Number(value)||0;return}
    if(key==='financeCustomMonths'){state.financeUI.customMonths=Number(value)||0;return}
    state.draft[key]=value;
    state.draft.updatedAt=new Date().toISOString();
  }

  function addComparable() {
    const name=$('comp-name')?.value.trim(),price=Number($('comp-price')?.value),area=Number($('comp-area')?.value);
    if(!name||!price||!area){toast('Nombre, precio y área son obligatorios');return}
    state.draft.comparables.push({id:uid(),name,price,area,rooms:Number($('comp-rooms')?.value)||0,date:$('comp-date')?.value||'',source:$('comp-source')?.value.trim()||'',url:$('comp-url')?.value.trim()||'',excluded:false});
    saveCurrent(true);render();toast('Comparable añadido');
  }

  function addRenovation() {
    const category=$('reno-category')?.value.trim(),qty=Number($('reno-qty')?.value),unitCost=Number($('reno-unit-cost')?.value);
    if(!category||!qty||!unitCost){toast('Partida, cantidad y coste unitario son obligatorios');return}
    state.draft.renovationItems.push({id:uid(),category,unit:$('reno-unit')?.value.trim()||'global',qty,unitCost,actualCost:$('reno-actual')?.value===''?'':Number($('reno-actual')?.value)||0,supplier:$('reno-supplier')?.value.trim()||'',status:$('reno-status')?.value||'Pendiente',notes:$('reno-notes')?.value.trim()||''});
    saveCurrent(true);render();toast('Partida añadida');
  }

  async function handlePhotos(event) {
    const files=[...event.target.files].slice(0,Math.max(0,6-(state.draft.photos||[]).length));
    for(const file of files){try{state.draft.photos.push(await compressImage(file))}catch(_){toast('No se pudo procesar una imagen')}}
    saveCurrent(true);render();
  }
