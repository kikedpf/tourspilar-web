      localStorage.setItem(STORAGE.compare, JSON.stringify([...state.compareIds]));
      if (state.currentId) localStorage.setItem(STORAGE.current, state.currentId);
    } catch (error) {
      toast('El almacenamiento está lleno. Exporta una copia y elimina fotos pesadas.');
    }
  }

  function toast(message) {
    const el = $('toast');
    el.textContent = message;
    el.classList.add('show');
    clearTimeout(toast.timer);
    toast.timer = setTimeout(() => el.classList.remove('show'), 2800);
  }

  function setCurrent(id) {
    const op = state.operations.find(item => item.id === id);
    if (!op) return;
    state.currentId = id;
    state.draft = clone(op);
    localStorage.setItem(STORAGE.current, id);
    refreshOperationSelect();
    render();
  }

  function saveCurrent(silent = false) {
    if (!state.draft) return;
    if (!state.draft.name.trim()) state.draft.name = `Operación ${new Date().toLocaleDateString('es-CO')}`;
    state.draft.updatedAt = new Date().toISOString();
    const index = state.operations.findIndex(o => o.id === state.draft.id);
    if (index >= 0) state.operations[index] = clone(state.draft);
    else state.operations.unshift(clone(state.draft));
    state.currentId = state.draft.id;
    persist();
    refreshOperationSelect();
    if (!silent) toast('Operación guardada');
  }

  function createNew() {
    state.draft = baseOperation();
    state.currentId = state.draft.id;
    state.operations.unshift(clone(state.draft));
    persist();
    refreshOperationSelect();
    navigate('analyzer');
    toast('Nueva operación creada');
  }

  function deleteOperation(id) {
    const op = state.operations.find(o => o.id === id);
    if (!op || !confirm(`¿Eliminar “${op.name}”?`)) return;
    state.operations = state.operations.filter(o => o.id !== id);
    state.compareIds.delete(id);
    if (!state.operations.length) state.operations.push(baseOperation());
    if (state.currentId === id) {
      state.currentId = state.operations[0].id;
      state.draft = clone(state.operations[0]);
    }
    persist();
    refreshOperationSelect();
    render();
    toast('Operación eliminada');
  }

  function duplicateOperation(id) {
    const op = state.operations.find(o => o.id === id);
    if (!op) return;
    const copy = clone(op);
    copy.id = uid();
    copy.name = `${op.name} · copia`;
    copy.status = 'Detectado';
    copy.isDemo = false;
    copy.createdAt = copy.updatedAt = new Date().toISOString();
    state.operations.unshift(copy);
    persist();
    toast('Operación duplicada');
    render();
  }

  function navigate(view) {
    state.view = view;
    qsa('.nav-item').forEach(btn => btn.classList.toggle('active', btn.dataset.view === view));
    const [title, subtitle] = VIEWS[view] || VIEWS.dashboard;
    $('page-title').textContent = title;
    $('page-subtitle').textContent = subtitle;
    $('sidebar').classList.remove('open');
    render();
  }

  function refreshOperationSelect() {
    const select = $('current-operation');
    select.innerHTML = state.operations.map(op => `<option value="${esc(op.id)}" ${op.id===state.currentId?'selected':''}>${esc(op.name)}</option>`).join('');
  }

  function calc(op = state.draft) { return E.calculate(op || baseOperation()); }
  function currentStored() { return state.operations.find(o => o.id === state.currentId); }
  function secondaryMoney(value) { return state.settings.showEur ? `<small>${eur(value)}</small>` : ''; }

  function render() {
    const root = $('view-root');
    const renderer = {
      dashboard: renderDashboard,
      analyzer: renderAnalyzer,
      opportunities: renderOpportunities,
      comparables: renderComparables,

      risk: renderRisk,
      pipeline: renderPipeline,
      reports: renderReports,
      settings: renderSettings
    }[state.view] || renderDashboard;
    root.innerHTML = renderer();
    bindDynamicAfterRender();
  }
