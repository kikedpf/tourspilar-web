  function breakdownHtml(r) {
    const rows = [
      ['Compra',r.breakdown.purchasePrice],['Gastos de compra',r.breakdown.purchaseCosts],['Reforma',r.breakdown.renovationTotal],['Comisión + IVA',r.breakdown.commissionTotal],['Gastos de venta',r.breakdown.saleCosts],['Costes fijos',r.breakdown.fixedCosts],['Contingencia',r.breakdown.contingency],['Intereses manuales',r.breakdown.manualInterest],['Otros costes',r.breakdown.otherCosts],['ICA',r.breakdown.ica],['Sobretasa bomberil',r.breakdown.fireSurcharge],['Coste de caja',r.breakdown.cashCost,true],['Coste económico total',r.breakdown.totalEconomicCost,true]
    ];
    return `<div class="breakdown">${rows.filter(x=>x[1]||x[2]).map(([label,value,total])=>`<div class="breakdown-row ${total?'total':''}"><span>${esc(label)}</span><strong>${money(value)}</strong></div>`).join('')}</div>`;
  }

  function field(label,key,value,type='text') { return `<label>${esc(label)}<input type="${type}" data-bind="${key}" value="${esc(value)}"></label>`; }

  function numberField(label,key,value,suffix='') { return `<label>${esc(label)}<span class="input-wrap"><input type="number" min="0" step="0.1" data-bind="${key}" data-kind="number" value="${Number(value)||0}">${suffix?`<span>${esc(suffix)}</span>`:''}</span></label>`; }

  function renderOpportunities() {
    const filtered = state.operations.filter(op => {
      const r = calc(op);
      const text = `${op.name} ${op.neighborhood} ${op.address}`.toLowerCase();
      return (!state.filters.search || text.includes(state.filters.search.toLowerCase())) && (!state.filters.status || op.status===state.filters.status) && (!state.filters.decision || r.actual.decision===state.filters.decision);
    }).sort((a,b)=>calc(b).score.total-calc(a).score.total);
    const selected = state.operations.filter(o=>state.compareIds.has(o.id)).map(op=>({op,r:calc(op)}));
    return `<article class="card">
      <div class="card-head"><div><span class="eyebrow">RADAR DE COMPRA</span><h2>${state.operations.length} operaciones guardadas</h2><p>Filtra, ordena y compara antes de gastar tiempo en visitas.</p></div><button class="primary-btn" data-action="new">Nueva oportunidad</button></div>
      <div class="toolbar no-print">
        <input class="grow" id="filter-search" type="search" placeholder="Buscar por operación, barrio o dirección" value="${esc(state.filters.search)}">
        <select id="filter-status"><option value="">Todos los estados</option>${STATUS.map(s=>`<option value="${esc(s)}" ${state.filters.status===s?'selected':''}>${esc(s)}</option>`).join('')}</select>

      </div>
      ${selected.length>=2 ? `<div class="recommendation" style="margin:16px 0"><strong>Comparación activa:</strong> ${selected.length} operaciones. Mejor ROI: ${esc([...selected].sort((a,b)=>b.r.actual.roi-a.r.actual.roi)[0].op.name)}.</div>`:''}

  }
