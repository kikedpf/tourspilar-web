  function load() {
    state.settings = readJSON(STORAGE.settings, defaultSettings());
    const stored = readJSON(STORAGE.operations, null);
    state.operations = Array.isArray(stored) && stored.length ? stored : [demoOperation()];
    state.compareIds = new Set(readJSON(STORAGE.compare, []));
    const preferred = localStorage.getItem(STORAGE.current);
    state.currentId = state.operations.some(o => o.id === preferred) ? preferred : state.operations[0].id;
    state.draft = clone(state.operations.find(o => o.id === state.currentId));
    const theme = localStorage.getItem(STORAGE.theme) || 'light';
    document.documentElement.dataset.theme = theme === 'dark' ? 'dark' : '';
    persist();
  }
