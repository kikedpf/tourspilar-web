  function financeChart(rows) {
    if (!rows.length) return '';
    const w=900,h=230,p=42,values=rows.map(r=>r.netProfit),min=Math.min(0,...values),max=Math.max(1,...values);
    const x=i=>p+(w-p*2)*(rows.length===1?.5:i/(rows.length-1));
    const y=v=>h-p-(v-min)/(max-min||1)*(h-p*2);
    const points=rows.map((r,i)=>`${x(i)},${y(r.netProfit)}`).join(' ');
    const grid=[0,.25,.5,.75,1].map(t=>{const yy=p+t*(h-p*2),val=max-t*(max-min);return `<line class="chart-grid" x1="${p}" y1="${yy}" x2="${w-p}" y2="${yy}"/><text class="chart-label" x="2" y="${yy+4}">${esc(compact(val))}</text>`}).join('');
    const dots=rows.map((r,i)=>`<circle class="chart-dot" cx="${x(i)}" cy="${y(r.netProfit)}" r="6"/><text class="chart-label" text-anchor="middle" x="${x(i)}" y="${h-8}">${r.months}m</text><text class="chart-label" text-anchor="middle" x="${x(i)}" y="${y(r.netProfit)-12}">${esc(compact(r.netProfit))}</text>`).join('');
    return `<svg viewBox="0 0 ${w} ${h}"><polyline class="chart-area" points="${p},${h-p} ${points} ${w-p},${h-p}"/><g>${grid}</g><polyline class="chart-line" points="${points}"/>${dots}</svg>`;
  }
